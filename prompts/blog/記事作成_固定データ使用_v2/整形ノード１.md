const parseOutput = (node) => {
  try {
    const raw = node.output ?? node.json ?? '{}';
    if (!raw || raw.trim() === '') throw new Error('outputが空です');
    const cleaned = raw
      .replace(/```json|```/g, '')
      .replace(/,(\s*[}\]])/g, '$1')  // トレイリングカンマ除去
      .trim();
    return JSON.parse(cleaned);
  } catch(e) { throw new Error(`JSONパース失敗: ${e.message}`); }
};
